212  class T,
213  class E,
214  class Timestamp,
215  class ReadOnlyPredicate>
216 bool BufferedLoggerOutput<Lock, MaxEntrySize, T, E, Timestamp, ReadOnlyPredicate>::outputEntry(
217  IEntryOutput<E, Timestamp>& output, EntryRefType& entryRef) const
218 {
219  uint8_t entryBuffer[MaxEntrySize];
220  uint32_t size;
221  {
222  Lock const lock;
223  size = _entryBuffer.getNextEntry(entryBuffer, entryRef);
224  }
225  if (size > 0U)
226  {
227  EntryOutputAdapter outputAdapter(_componentMapping, entryRef.getIndex(), output);
228 
229  _entrySerializer.deserialize(::etl::span<uint8_t>(entryBuffer).first(size), outputAdapter);
230  }
231  return size > 0U;
232 }
