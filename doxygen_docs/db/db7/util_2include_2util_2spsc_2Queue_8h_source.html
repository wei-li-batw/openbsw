42  // The tx.sent and rx.received variables are always in the range [0, 2*N] while the ring
43  // contains N elements. This way there are two distinct constellations where
44  // tx.sent and rx.received point two the same element:
45  // 1) tx.sent == rx.received
46  // 2) tx.sent == rx.received + N
47  // Using this trick the ambiguity between the empty and full cases of the queue is resolved
48  // without the need for an unused element in the queue: Case 1) means "empty" and case 2) means
49  // "full".

189 template<class T, size_t N, bool>
190 struct TxData
191 {
192  ::etl::atomic<size_t> sent;
193  ::etl::atomic<size_t> acked;
194  ::etl::array<T, N> data;
195 
196  TxData() : sent(0U), acked(0U), data() {}
197 
198  void reset()
199  {
200  sent.store(0U);
201  acked.store(0U);
202  }
203 
204  template<class RxData, class OnConsumed>
205  void check(RxData const& rxData, OnConsumed const& onConsumed)
206  {
207  size_t numAcked = acked.load();
208  while (numAcked != rxData.received.load())
209  {
210  onConsumed(data[numAcked % N]);
211  numAcked = (numAcked + 1U) % (2U * N);
212  acked.store(numAcked);
213  }
214  }
215 
216  template<class RxData>
217  bool full(RxData const& /*rxData*/) const
218  {
219  size_t const numSent = sent.load();
220  size_t const numAcked = acked.load();
221  return (numSent == ((numAcked + N) % (2U * N)));
222  }
223 };
