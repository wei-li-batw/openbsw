31 inline ::etl::span<uint8_t> align(size_t const alignment, ::etl::span<uint8_t> s)
32 {
33  auto const mod = reinterpret_cast<size_t>(s.data()) % alignment;
34  if (mod != 0)
35  {
36  return s.subspan(alignment - mod);
37  }
38  else
39  {
40  return s;
41  }
42 }

259 template<class Adapter>
260 template<typename T>
261 void TaskInitializer<Adapter>::create(
262  ContextType const context,
263  char const* const name,
264  TaskObjectType& task,
265  T& stack,
266  TaskFunctionType const taskFunction,
267  TaskConfigType const& config)
268 {
269  auto const stackSliceRaw = ::etl::make_span(stack).template reinterpret_as<uint8_t>();
270  auto const stackSlice
271  = align(alignof(StackType_t), stackSliceRaw).template reinterpret_as<StackType_t>();
272  auto const taskInitializerSlice = align(alignof(TaskInitializer), stackSliceRaw);
273  ETL_ASSERT(
274  taskInitializerSlice.size() >= sizeof(TaskInitializer),
275  ETL_ERROR_GENERIC("stack must be big enough to hold a task initializer object"));
276  new (taskInitializerSlice.data())
277  TaskInitializer(context, name, task, stackSlice, taskFunction, config);
278 }
