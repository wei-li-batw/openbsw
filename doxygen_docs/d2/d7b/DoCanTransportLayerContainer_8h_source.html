183 template<class DataLinkLayer>
184 void DoCanTransportLayerContainer<DataLinkLayer>::shutdown(
185  ShutdownCallbackType const shutdownCallback)
186 {
187  _shutdownPendingCount = 1U;
188  _shutdownCallback = shutdownCallback;
189  for (TransportLayerType& layer : _layers)
190  {
191  {
192  // Even if the value changes between now and the increment, we want to ensure this
193  // function is never about to be incremented when _shutdownPendingCount is ever the max
194  // possible value.
195  ETL_ASSERT(
196  _shutdownPendingCount
197  != ::etl::numeric_limits<decltype(_shutdownPendingCount)>::max(),
198  ETL_ERROR_GENERIC("pending count must not wrap"));
199  ::interrupts::SuspendResumeAllInterruptsScopedLock const lock;
200  ++_shutdownPendingCount;
201  }
202  (void)layer.shutdown(ShutdownDelegate::create<
203  DoCanTransportLayerContainer<DataLinkLayer>,
204  &DoCanTransportLayerContainer<DataLinkLayer>::shutdownDone>(*this));
205  }
206  singleShutdownDone();
207 }

239 template<class DataLinkLayer>
240 void DoCanTransportLayerContainer<DataLinkLayer>::singleShutdownDone()
241 {
242  bool done;
243  {
244  // Even if the value changes between now and the decrement, we want to ensure this function
245  // is never called when _shutdownPendingCount is ever 0.
246  ETL_ASSERT(_shutdownPendingCount != 0, ETL_ERROR_GENERIC("pending count must not wrap"));
247  ::interrupts::SuspendResumeAllInterruptsScopedLock const lock;
248  --_shutdownPendingCount;
249  done = (_shutdownPendingCount == 0U);
250  }
251  if (done)
252  {
253  _shutdownCallback();
254  }
255 }
