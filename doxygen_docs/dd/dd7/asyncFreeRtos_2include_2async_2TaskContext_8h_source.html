302 template<class Binding>
303 inline EventMaskType TaskContext<Binding>::waitEvents()
304 {
305  EventMaskType eventMask = 0U;
306  uint32_t ticks = Binding::WAIT_EVENTS_TICK_COUNT;
307  uint32_t nextDelta;
308  bool const hasDelta = _timer.getNextDelta(getSystemTimeUs32Bit(), nextDelta);
309  if (hasDelta)
310  {
311  ticks = static_cast<uint32_t>((nextDelta + (Config::TICK_IN_US - 1U)) / Config::TICK_IN_US);
312  }
313  if (xTaskNotifyWait(0U, WAIT_EVENT_MASK, &eventMask, ticks) != 0)
314  {
315  return eventMask;
316  }
317  else if (hasDelta)
318  {
319  return TimerEventPolicyType::EVENT_MASK;
320  }
321  else
322  {
323  return 0U;
324  }
325 }
