28  // PRESDIV = 39 -> fTQ = fCLK / 40 = 1 MHz

30  CTRL_TIMING_LOWSPEED = 0x27610001,
31  // PRESDIV = 7 -> fTQ = fCLK / 8 = 5 MHz

33  CTRL_TIMING_HIGHSPEED = 0x07610001,
34  // PRESDIV = 3 -> fTQ = fCLK / 4 = 10 MHz

263  struct MessageBuffer8Byte
264  {
265  union
266  { // FLAGS Register
267  vuint32_t R;
268 
269  struct
270  {
271  // Free running timestamp.
272  vuint32_t TIMESTAMP : 16;
273  // Length of data.
274  vuint32_t DLC : 4;
275  // Remote Transmission Request.
276  vuint32_t RTR : 1;
277  // ID Extended Bit. 1 extended, 0 standard.
278  vuint32_t IDE : 1;
279  // Substitute Remote Request
280  vuint32_t SRR : 1;
281  vuint32_t unnamed0 : 1;
282  vuint32_t CODE : 4;
283  vuint32_t unnamed1 : 1;
284  // Error State Indicator. Indicates if tx node is in error.
285  vuint32_t ESI : 1;
286  // Bit Rate Switch. Defines bit rate switch for CAN FD.
287  vuint32_t BRS : 1;
288  // Extended Data Length. Distinguishes between CAN and CAN FD.
289  vuint32_t EDL : 1;
290  } B;
291  } FLAGS;
292 
293  union
294  { // ID Register
295  vuint32_t R;
296 
297  struct
298  {
299  vuint32_t ID_EXT : 18;
300  vuint32_t ID_STD : 11;
301  vuint32_t PRIO : 3;
302  } B;
303  } ID;
304 
305  union
306  { // Frame payload
307  vuint8_t B[8];
308  vuint16_t H[4];
309  vuint32_t W[2];
310  } DATA;
311  };
