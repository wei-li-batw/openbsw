139 template<class DataLinkLayer, class AddressEntry>
140 void DoCanNormalAddressingFilter<DataLinkLayer, AddressEntry>::init(
141  AddressEntrySliceType const addressEntries, CodecsSliceType const codecEntries)
142 {
143  // Address entries must be non-zero in size
144  ETL_ASSERT(
145  addressEntries.size() != 0, ETL_ERROR_GENERIC("list of addresses must not be empty"));
146 
147  _entries = addressEntries;
148  _codecEntries = codecEntries;
149  typename AddressEntrySliceType::iterator firstExtendedEntryTypeIt = _entries.end();
150  uint32_t prevCanReceptionId = 0U;
151  typename AddressEntrySliceType::const_iterator it = _entries.begin();
152 
153  for (; it != _entries.end(); ++it)
154  {
155  DataLinkAddressType const canReceptionId = it->_canReceptionId;
156  if (it != _entries.begin())
157  {
158  ETL_ASSERT(
159  prevCanReceptionId < canReceptionId,
160  ETL_ERROR_GENERIC("can reception id must be monotonically increasing"));
161  }
162  prevCanReceptionId = canReceptionId;
163  if (::can::CanId::isBase(canReceptionId))
164  {
165  BitFieldFilter::add(canReceptionId);
166  }
167  else
168  {
169  if (firstExtendedEntryTypeIt == _entries.end())
170  {
171  firstExtendedEntryTypeIt = it;
172  }
173  if (!::can::CanId::isValid(canReceptionId))
174  {
175  break;
176  }
177  }
178  }
179 
180  _extendedEntries = AddressEntrySliceType{
181  firstExtendedEntryTypeIt, static_cast<size_t>(it - firstExtendedEntryTypeIt)};
182 
183  for (; it != _entries.end(); ++it)
184  {
185  ETL_ASSERT(
186  !::can::CanId::isValid(it->_canReceptionId),
187  ETL_ERROR_GENERIC("can reception id must be valid"));
188  }
189 }

191 template<class DataLinkLayer, class AddressEntry>
192 typename DoCanNormalAddressingFilter<DataLinkLayer, AddressEntry>::FrameCodecType const*
193 DoCanNormalAddressingFilter<DataLinkLayer, AddressEntry>::getTransmissionParameters(
194  DoCanTransportAddressPair const& transportAddressPair,
195  DataLinkAddressPairType& dataLinkAddressPair) const
196 {
197  uint16_t const sourceId = transportAddressPair.getTargetId();
198  uint16_t const targetId = transportAddressPair.getSourceId();
199  for (AddressEntryType const& entry : _entries)
200  {
201  if ((entry._transportSourceId == sourceId) && (entry._transportTargetId == targetId))
202  {
203  dataLinkAddressPair
204  = DataLinkAddressPairType(entry._canReceptionId, entry._canTransmissionId);
205  return getFrameCodec(entry._transmissionCodecIdx);
206  }
207  }
208  return nullptr;
209 }

211 template<class DataLinkLayer, class AddressEntry>
212 typename DoCanNormalAddressingFilter<DataLinkLayer, AddressEntry>::FrameCodecType const*
213 DoCanNormalAddressingFilter<DataLinkLayer, AddressEntry>::getReceptionParameters(
214  DataLinkAddressType receptionAddress,
215  DoCanTransportAddressPair& transportAddressPair,
216  DataLinkAddressType& transmissionAddress) const
217 {
218  // search all entries from first base entry to last extended entry.
219  AddressEntryType const* const entry = findEntryTypeByReceptionAddress(
220  receptionAddress, _entries.begin(), _extendedEntries.end());
221  if (entry == nullptr)
222  {
223  return nullptr;
224  }
225 
226  transmissionAddress = entry->_canTransmissionId;
227  transportAddressPair
228  = DoCanTransportAddressPair(entry->_transportSourceId, entry->_transportTargetId);
229  return getFrameCodec(entry->_receptionCodecIdx);
230 }

255 template<class DataLinkLayer, class AddressEntry>
256 typename DoCanNormalAddressingFilter<DataLinkLayer, AddressEntry>::AddressEntryType const*
257 DoCanNormalAddressingFilter<DataLinkLayer, AddressEntry>::findEntryTypeByReceptionAddress(
258  DataLinkAddressType const receptionAddress,
259  typename AddressEntrySliceType::const_iterator const beginIt,
260  typename AddressEntrySliceType::const_iterator const endIt)
261 {
262  AddressEntryType key{};
263  key._canReceptionId = receptionAddress;
264  typename AddressEntrySliceType::const_iterator const it
265  = ::etl::lower_bound(beginIt, endIt, key, &lessReceptionId);
266  return ((it != endIt) && (it->_canReceptionId == receptionAddress)) ? &*it : nullptr;
267 }
