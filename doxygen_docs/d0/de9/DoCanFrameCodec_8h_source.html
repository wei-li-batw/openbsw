211 {

213  if (payload.size() <= offset)

222 template<class DataLinkLayer>
223 CodecResult DoCanFrameCodec<DataLinkLayer>::decodeSingleFrame(
224  ::etl::span<uint8_t const> const& payload,
225  MessageSizeType& messageSize,
226  ::etl::span<uint8_t const>& data) const
227 {
228  size_t const offset = _config._offset;
229  if (checkFrameSize(payload, 2U, _config._singleFrameSize))
230  {
231  messageSize = static_cast<uint8_t>(payload[offset] & 0x0fU);
232  bool const isExtendedSfDlExpected = payload.size() > EXTENDED_SF_DL_EDGE_SIZE;
233  if ((messageSize == 0U) && isExtendedSfDlExpected)
234  {
235  messageSize = payload[offset + 1U];
236  MessageSizeType const checkPayloadSize
237  = messageSize + static_cast<MessageSizeType>(offset) + 2U;
238 
239  if ((checkPayloadSize <= static_cast<MessageSizeType>(_config._singleFrameSize._max))
240  && (static_cast<size_t>(checkPayloadSize) <= payload.size()))
241  {
242  data = ::etl::span<uint8_t const>(
243  &payload[offset + 2U], static_cast<size_t>(messageSize));
244  return CodecResult::OK;
245  }
246  }
247  else if ((messageSize > 0U) && (!isExtendedSfDlExpected))
248  {
249  MessageSizeType const checkPayloadSize
250  = messageSize + static_cast<MessageSizeType>(offset) + 1U;
251  if (checkPayloadSize <= payload.size())
252  {
253  data = ::etl::span<uint8_t const>(
254  &payload[offset + 1U], static_cast<size_t>(messageSize));
255  return CodecResult::OK;
256  }
257  }
258  else
259  {
260  // invalid message size
261  }
262  return CodecResult::INVALID_MESSAGE_SIZE;
263  }
264  return CodecResult::INVALID_FRAME_SIZE;
265 }

267 template<class DataLinkLayer>
268 CodecResult DoCanFrameCodec<DataLinkLayer>::decodeFirstFrame(
269  ::etl::span<uint8_t const> const& payload,
270  MessageSizeType& messageSize,
271  FrameIndexType& frameCount,
272  FrameSizeType& consecutiveFrameDataSize,
273  ::etl::span<uint8_t const>& data) const
274 {
275  if (checkFrameSize(payload, 3U, _config._firstFrameSize))
276  {
277  size_t const offset = _config._offset;
278  messageSize = static_cast<MessageSizeType>(etl::be_uint16_t(&payload[offset]) & 0xFFFU);
279  uint8_t dataStart = 2U;
280  consecutiveFrameDataSize = static_cast<FrameSizeType>(payload.size())
281  - (static_cast<FrameSizeType>(offset) + 1U);
282  if (messageSize == 0U)
283  {
284  if (!checkFrameSize(payload, 6U, _config._firstFrameSize))
285  {
286  return CodecResult::INVALID_FRAME_SIZE;
287  }
288  // Escape sequence, aka message size > 4095 up to uint32_t
289  uint32_t const escapedMessageSize = etl::be_uint32_t(&payload[offset + 2U]);
290 
291  if ((escapedMessageSize <= ESCAPED_SEQ_MESSAGE_SIZE)
292  || (escapedMessageSize > ::etl::numeric_limits<MessageSizeType>::max()))
293  {
294  return CodecResult::INVALID_MESSAGE_SIZE;
295  }
296  messageSize = static_cast<MessageSizeType>(escapedMessageSize);
297  dataStart = 6U;
298  }
299 
300  MessageSizeType const count = (messageSize / consecutiveFrameDataSize);
301  if (count >= etl::numeric_limits<FrameIndexType>::max())
302  {
303  return CodecResult::INVALID_MESSAGE_SIZE;
304  }
305  frameCount = static_cast<FrameIndexType>(count + 1U);
306  bool const msgFitsIntoLongSingleFrame = fitsIntoLongSingleFrame(messageSize);
307  bool const msgFitsIntoShortSingleFrame = fitsIntoShortSingleFrame(messageSize);
308  if ((frameCount > 1U) && (!msgFitsIntoShortSingleFrame) && (!msgFitsIntoLongSingleFrame))
309  {
310  data = ::etl::span<uint8_t const>(
311  &payload[offset + dataStart],
312  static_cast<size_t>(consecutiveFrameDataSize) - dataStart + 1U);
313  return CodecResult::OK;
314  }
315  return CodecResult::INVALID_MESSAGE_SIZE;
316  }
317  return CodecResult::INVALID_FRAME_SIZE;
318 }

356  MessageSizeType const messageSize) const
357 {
358  FrameSizeType const edge = ::etl::min(EXTENDED_SF_DL_EDGE_SIZE, _config._singleFrameSize._max);

360  return messageSize <= static_cast<MessageSizeType>(edge) - offset;

363 template<class DataLinkLayer>
364 bool DoCanFrameCodec<DataLinkLayer>::fitsIntoLongSingleFrame(
365  MessageSizeType const messageSize) const
366 {
367  // Ensure _singleFrameSize._max is at least as large as _config._offset + 2
368  ETL_ASSERT(
369  _config._singleFrameSize._max >= _config._offset + 2,
370  ETL_ERROR_GENERIC("single frame size must be large enough"));
371  uint8_t const offset = _config._offset + 2U;
372  return messageSize <= static_cast<MessageSizeType>(_config._singleFrameSize._max) - offset;
373 }

375 template<class DataLinkLayer>
376 CodecResult DoCanFrameCodec<DataLinkLayer>::getEncodedFrameCount(
377  MessageSizeType const messageSize,
378  FrameIndexType& frameCount,
379  FrameSizeType& consecutiveFrameDataSize) const
380 {
381  // The max consecutive frame size needs to be at least 2 larger than the configured offset for
382  // this function.
383  ETL_ASSERT(
384  _config._consecutiveFrameSize._max >= _config._offset + 2,
385  ETL_ERROR_GENERIC("consecutive frame size must be large enough"));
386 
387  if (messageSize == 0U)
388  {
389  return CodecResult::INVALID_MESSAGE_SIZE;
390  }
391 
392  bool const msgFitsIntoLongSingleFrame = fitsIntoLongSingleFrame(messageSize);
393  bool const msgFitsIntoShortSingleFrame = fitsIntoShortSingleFrame(messageSize);
394  if (msgFitsIntoShortSingleFrame || msgFitsIntoLongSingleFrame)
395  {
396  consecutiveFrameDataSize = 0U;
397  frameCount = 1U;
398  }
399  else
400  {
401  consecutiveFrameDataSize = _config._consecutiveFrameSize._max - (_config._offset + 1U);
402  MessageSizeType const count = (messageSize / consecutiveFrameDataSize);
403  if (count >= etl::numeric_limits<FrameIndexType>::max())
404  {
405  return CodecResult::INVALID_MESSAGE_SIZE;
406  }
407  frameCount = static_cast<FrameIndexType>(count + 1U);
408  }
409  return CodecResult::OK;
410 }

412 template<class DataLinkLayer>
413 CodecResult DoCanFrameCodec<DataLinkLayer>::encodeDataFrame(
414  ::etl::span<uint8_t>& payload,
415  ::etl::span<uint8_t const> const& data,
416  FrameIndexType const frameIndex,
417  FrameSizeType const consecutiveFrameDataSize,
418  FrameSizeType& consumedDataSize) const
419 {
420  if ((data.size() == 0U) || (data.size() > ::etl::numeric_limits<MessageSizeType>::max()))
421  {
422  return CodecResult::INVALID_FRAME_INDEX;
423  }
424  MessageSizeType const pendingMessageSize = static_cast<MessageSizeType>(data.size());
425 
426  FrameSizeType destOffset;
427  FrameSizeType minFrameSize = 0U;
428  if (frameIndex == 0U)
429  {
430  size_t const offset = _config._offset;
431  if (fitsIntoShortSingleFrame(pendingMessageSize))
432  {
433  payload[offset] = static_cast<uint8_t>(FrameType::SINGLE) << 4U;
434  payload[offset] |= static_cast<uint8_t>(pendingMessageSize) & 0x0FU;
435 
436  destOffset = static_cast<FrameSizeType>(offset) + 1U;
437  consumedDataSize = static_cast<FrameSizeType>(pendingMessageSize);
438  minFrameSize = _config._singleFrameSize._min;
439  }
440  else if (fitsIntoLongSingleFrame(pendingMessageSize))
441  {
442  payload[offset] = static_cast<uint8_t>(FrameType::SINGLE) << 4U;
443  payload[offset + 1U] = static_cast<uint8_t>(pendingMessageSize);
444  destOffset = static_cast<FrameSizeType>(offset) + 2U;
445  consumedDataSize = static_cast<FrameSizeType>(pendingMessageSize);
446  minFrameSize = _config._singleFrameSize._min;
447  }
448  else if (consecutiveFrameDataSize > 0U)
449  {
450  destOffset = static_cast<FrameSizeType>(offset) + 2U;
451  consumedDataSize = consecutiveFrameDataSize - 1U;
452 
453  if (pendingMessageSize <= ESCAPED_SEQ_MESSAGE_SIZE)
454  {
455  etl::be_uint16_ext_t{&payload[offset]} = pendingMessageSize & 0xFFFU;
456  }
457  else
458  {
459  payload[offset] = 0U;
460  payload[offset + 1U] = 0U;
461  etl::be_uint32_ext_t{&payload[offset + 2U]} = pendingMessageSize;
462 
463  destOffset += 4U;
464  consumedDataSize -= 4U;
465  }
466  payload[offset] |= static_cast<uint8_t>(FrameType::FIRST) << 4U;
467  minFrameSize = _config._firstFrameSize._min;
468  }
469  else
470  {
471  return CodecResult::INVALID_FRAME_INDEX;
472  }
473  }
474  else if (consecutiveFrameDataSize > 0U)
475  {
476  uint8_t const offset = _config._offset;
477 
478  payload[offset] = static_cast<uint8_t>(FrameType::CONSECUTIVE) << 4U;
479  payload[offset] |= static_cast<uint8_t>(frameIndex) & 0x0FU;
480 
481  destOffset = offset + 1U;
482  consumedDataSize = static_cast<FrameSizeType>(
483  ::etl::min(static_cast<MessageSizeType>(consecutiveFrameDataSize), pendingMessageSize));
484  minFrameSize = ::etl::max(consumedDataSize, _config._consecutiveFrameSize._min);
485  }
486  else
487  {
488  return CodecResult::INVALID_FRAME_INDEX;
489  }
490 
491  FrameSizeType const payloadSize = (destOffset + consumedDataSize);
492  if (payloadSize > payload.size())
493  {
494  return CodecResult::INVALID_FRAME_SIZE;
495  }
496 
497  (void)::etl::mem_copy(
498  &data[0U],
499  static_cast<size_t>(consumedDataSize),
500  &payload[static_cast<size_t>(destOffset)]);
501  return adjustFrame(payload, payloadSize, minFrameSize);
502 }

504 template<class DataLinkLayer>
505 CodecResult DoCanFrameCodec<DataLinkLayer>::encodeFlowControlFrame(
506  ::etl::span<uint8_t>& payload,
507  FlowStatus const flowStatus,
508  uint8_t const blockSize,
509  uint8_t const encodedMinSeparationTime) const
510 {
511  size_t const offset = _config._offset;
512  if (payload.size() < (offset + 3U))
513  {
514  return CodecResult::INVALID_FRAME_SIZE;
515  }
516 
517  payload[offset] = static_cast<uint8_t>(FrameType::FLOW_CONTROL) << 4U;
518  payload[offset] |= static_cast<uint8_t>(flowStatus) & 0x0FU;

522  return adjustFrame(
523  payload, static_cast<FrameSizeType>(offset) + 3U, _config._flowControlFrameSize._min);

526 template<class DataLinkLayer>
527 CodecResult DoCanFrameCodec<DataLinkLayer>::adjustFrame(
528  ::etl::span<uint8_t>& payload,
529  FrameSizeType const payloadSize,
530  FrameSizeType const minFrameSize) const
531 {
532  FrameSizeType paddedFrameSize = ::etl::max(payloadSize, minFrameSize);
533  if (!_mapper.mapFrameSize(paddedFrameSize))
534  {
535  return CodecResult::INVALID_FRAME_SIZE;
536  }
537 
538  if (paddedFrameSize > payloadSize)
539  {
540  if (paddedFrameSize > payload.size())
541  {
542  return CodecResult::INVALID_FRAME_SIZE;
543  }
544 
545  ::etl::mem_set(
546  &payload[static_cast<size_t>(payloadSize)],
547  static_cast<size_t>(paddedFrameSize) - payloadSize,
548  _config._filler);
549  }
550  payload = payload.first(static_cast<size_t>(paddedFrameSize));
551  return CodecResult::OK;
552 }
