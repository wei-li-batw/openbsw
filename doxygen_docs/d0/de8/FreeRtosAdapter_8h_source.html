305 template<class Binding>
306 void FreeRtosAdapter<Binding>::initTask(TaskInitializer& initializer)
307 {
308  ContextType const context = initializer._context;
309  _stackSizes[static_cast<size_t>(context)] = static_cast<uint32_t>(
310  static_cast<size_t>(initializer._stack.size()) * sizeof(BaseType_t));
311  TaskConfigsType::setTaskConfig(context, initializer._config);
312  if (context == TASK_TIMER)
313  {
314  _timerTaskInitializer = &initializer;
315  _timerTaskName = initializer._name;
316  }
317  else
318  {
319  TaskContextType& taskContext = _taskContexts[static_cast<size_t>(context)];
320  if (context == TASK_IDLE)
321  {
322  _idleTaskInitializer = &initializer;
323  taskContext.initTask(TASK_IDLE, initializer._name, initializer._taskFunction);
324  }
325  else
326  {
327  taskContext.createTask(
328  context,
329  initializer._task,
330  initializer._name,
331  static_cast<UBaseType_t>(context),
332  initializer._stack,
333  initializer._taskFunction);
334  }
335  }
