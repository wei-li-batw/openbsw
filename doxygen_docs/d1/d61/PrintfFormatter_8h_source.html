251 template<class T>
252 inline char* PrintfFormatter::formatIntDigits(
253  char* bufferEnd,
254  char const* const digits,
255  T value,
256  bool const signedType,
257  typename ::etl::make_unsigned<T>::type const base,
258  int8_t& sign)
259 {
260  if (value != 0U)
261  {
262  if (signedType)
263  {
264  if (static_cast<typename ::etl::make_signed<T>::type>(value) < 0)
265  {
266  value = -value;
267  sign = -1;
268  }
269  else
270  {
271  sign = +1;
272  }
273  }
274  else
275  {
276  sign = +1;
277  }
278 
279  while (value > 0U)
280  {
281  --bufferEnd;
282  *bufferEnd = digits[value % base];
283  value /= base;
284  }
285  }
286  else
287  {
288  sign = 0;
289  --bufferEnd;
290  *bufferEnd = '0';
291  }
292  return bufferEnd;
293 }

296 inline char const* PrintfFormatter::getIntSign(ParamInfo const& paramInfo, int8_t const sign)
297 {
298  char const* pSign = "";
299  if (paramInfo._base == 10U)
300  {
301  switch (paramInfo._datatype)
302  {
303  case ParamDatatype::SINT8:
304  case ParamDatatype::SINT16:
305  case ParamDatatype::SINT32:
306  case ParamDatatype::SINT64:
307  {
308  if (sign < 0)
309  {
310  pSign = "-";
311  }
312  else if ((paramInfo._flags & ParamFlags::FLAG_PLUS) > 0U)
313  {
314  pSign = "+";
315  }
316  else if ((paramInfo._flags & ParamFlags::FLAG_SPACE) > 0U)
317  {
318  pSign = " ";
319  }
320  else
321  {
322  // do nothing
323  }
324  break;
325  }
326  default:
327  {
328  break;
329  }
330  }
331  }
332  return pSign;

337 {
338  char const* pPrefix = "";
339  switch (paramInfo._base)
340  {
341  case 8:
342  {
343  if ((sign > 0) && ((paramInfo._flags & ParamFlags::FLAG_ALT) != 0U))
344  {
345  pPrefix = "0";
346  }
347  break;
348  }
349  case 16:
350  {
351  if ((paramInfo._flags & ParamFlags::FLAG_ALT) != 0U)
352  {
353  pPrefix = ((paramInfo._flags & ParamFlags::FLAG_UPPER) > 0U) ? "0X" : "0x";
354  }
355  break;
356  }
357  default:
358  {
359  break;
360  }
361  }
362  return pPrefix;
