133 template<class T, class Timestamp, class ReadOnlyPredicate>
134 T EntrySerializer<T, Timestamp, ReadOnlyPredicate>::serialize(
135  ::etl::span<uint8_t> const& destBuffer,
136  Timestamp const timestamp,
137  uint8_t const componentIndex,
138  ::util::logger::Level const level,
139  char const* const formatString,
140  va_list ap) const
141 {
142  using ::util::format::PrintfArgumentReader;
143  using ::util::format::PrintfFormatScanner;
144  uint8_t* const dest = destBuffer.data();
145 
146  EntryWriter writer(dest, dest + destBuffer.size(), _readOnlyPredicate);
147  writer.writeBytes(&timestamp, static_cast<T>(sizeof(timestamp)));
148  writer.writeBytes(&componentIndex, static_cast<T>(sizeof(componentIndex)));
149  uint8_t const levelByte = static_cast<uint8_t>(level);
150  writer.writeBytes(&levelByte, static_cast<T>(sizeof(levelByte)));
151  writer.writeString(formatString);
152  PrintfArgumentReader argReader(ap);
153 
154  for (PrintfFormatScanner scanner(formatString); scanner.hasToken(); scanner.nextToken())
155  {
156  if (scanner.getTokenType() == ::util::format::TokenType::PARAM)
157  {
158  if (scanner.getParamInfo()._width == ::util::format::ParamWidthOrPrecision::PARAM)
159  {
160  writer.copyParamVariant(argReader, ::util::format::ParamDatatype::SINT32);
161  }
162  if (scanner.getParamInfo()._precision == ::util::format::ParamWidthOrPrecision::PARAM)
163  {
164  writer.copyParamVariant(argReader, ::util::format::ParamDatatype::SINT32);
165  }
166 
167  writer.copyParamVariant(argReader, scanner.getParamInfo()._datatype);
168  }
169  }
170  return writer.getSize();
171 }

174 void EntrySerializer<T, Timestamp, ReadOnlyPredicate>::deserialize(
175  ::etl::span<uint8_t const> const& srcBuffer, IEntrySerializerCallback<Timestamp>& callback)
176 {
177  EntryReader reader(srcBuffer.data(), srcBuffer.data() + srcBuffer.size());
178  Timestamp timestamp;
179  uint8_t componentIdx;
180  uint8_t level;
181  reader.readBytes(&timestamp, static_cast<T>(sizeof(timestamp)));
182  reader.readBytes(&componentIdx, static_cast<T>(sizeof(componentIdx)));
183  reader.readBytes(&level, static_cast<T>(sizeof(level)));
184  ::util::format::ParamVariant const* const formatArg
185  = reader.readArgument(::util::format::ParamDatatype::CHARPTR);
186  if (formatArg != nullptr)
187  {
188  callback.onEntry(
189  timestamp,
190  componentIdx,
191  static_cast<::util::logger::Level>(level),
192  formatArg->_charPtrValue,
193  reader);
194  }
195 }

311 template<class T, class Timestamp, class ReadOnlyPredicate>
312 void EntrySerializer<T, Timestamp, ReadOnlyPredicate>::EntryWriter::writeString(
313  uint8_t const typeByte, char const* const source, T length)
314 {
315  writeBytes(&typeByte, static_cast<T>(sizeof(typeByte)));
316  T const completeLength = length + 1U;
317  T const headerLength = static_cast<T>(sizeof(T));
318 
319  if ((_write + (headerLength + completeLength)) > _bufferEnd)
320  {
321  // Add a constant length zero-terminated string to indicate overflow ("<?>" = 4 bytes)
322  if ((_write + (headerLength + 5U)) <= _bufferEnd)
323  {
324  length = static_cast<T>(_bufferEnd - _write) - static_cast<T>(headerLength + 4U);
325  writeBytes(&length, static_cast<T>(sizeof(length)));
326  writeBytes(source, length);
327  writeBytes("<?>", 4U);
328  return;
329  }
330  }
331  writeBytes(&completeLength, static_cast<T>(sizeof(completeLength)));
332  writeBytes(source, length);
333  uint8_t const zeroByte = 0U;
334  writeBytes(&zeroByte, static_cast<T>(sizeof(zeroByte)));
335 }

408  break;

410  case DATATYPE_SINT16:

412  {
413  readBytes(&_variant._uint16Value, static_cast<T>(sizeof(_variant._uint16Value)));
414  break;
415  }
416  case DATATYPE_SINT32:
417  case DATATYPE_UINT32:
418  {
419  readBytes(&_variant._uint32Value, static_cast<T>(sizeof(_variant._uint32Value)));
420  break;
421  }
422  case DATATYPE_SINT64:
423  case DATATYPE_UINT64:
424  {
425  readBytes(&_variant._uint64Value, static_cast<T>(sizeof(_variant._uint64Value)));
426  break;
427  }
428  case DATATYPE_VOIDPTR:
429  {
430  readBytes(&_variant._voidPtrValue, static_cast<T>(sizeof(_variant._voidPtrValue)));
431  break;
432  }
433  case DATATYPE_CHARPTR:
434  {
435  readBytes(&_variant._charPtrValue, static_cast<T>(sizeof(_variant._charPtrValue)));
436  break;
437  }
438  case DATATYPE_CHARARRAY:
439  {
440  _variant._charPtrValue = readCharArray();
441  break;
442  }
443  case DATATYPE_SIZEDCHARPTR:

448  _bufferEnd - reinterpret_cast<uint8_t const*>(_plainSizedString._data));
449  _plainSizedString._length = ::etl::strlen(_plainSizedString._data, limit);
450  break;
451  }
452  default:

455  break;
456  }
