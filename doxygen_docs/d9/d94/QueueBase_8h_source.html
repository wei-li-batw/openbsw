114  void takeSnapshot()
115  {
116  uint32_t const currentSize = size();
117  if (0U != currentSize)
118  {
119  _stats.loadSnapshot += static_cast<uint16_t>(currentSize);
120  ++_stats.processingCounter;
121  }
122  _stats.realLoadSnapshot += static_cast<uint16_t>(currentSize);
123  ++_stats.realProcessingCounter;
124  if (_stats.previousSnapshot == 0U)
125  {
126  _stats.maxFillRate = static_cast<uint8_t>(currentSize);
127  _stats.previousSnapshot = static_cast<uint8_t>(currentSize);
128  }
129  else
130  {
131  if (currentSize > _stats.previousSnapshot)
132  {
133  auto const diff = (currentSize - _stats.previousSnapshot);
134  if (diff > _stats.maxFillRate)
135  {
136  _stats.maxFillRate = static_cast<uint8_t>(diff);
137  }
138  }

140  }
