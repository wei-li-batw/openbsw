18 template<
19  class DigestType,
20  bool ReflectInput = false,
21  bool ReflectOutput = false,
22  DigestType FinalXor = 0>
23 struct CrcCalculatorBase
24 {
25  static DigestType update(
26  DigestType const (&lookupTable)[256],
27  uint8_t const* const data,
28  size_t const length,
29  DigestType const initValue = DigestType())
30  {
31  DigestType reg = initValue;
32  ETL_ASSERT(data != nullptr || length == 0, ETL_ERROR_GENERIC("data must not be null"));
33  for (size_t idx = 0U; idx < length; ++idx)
34  {
35  uint8_t const value = Reflect<ReflectInput>::apply(data[idx]);
36  uint8_t const lookupTableIndex = value ^ (reg >> ((sizeof(DigestType) * 8U) - 8U));
37  reg = lookupTable[lookupTableIndex] ^ static_cast<DigestType>(reg << 8U);
38  }
39  return reg;
40  }
41 
42  static DigestType digest(DigestType const value)
43  {
44  return Xor<DigestType, FinalXor>::apply(Reflect<ReflectOutput>::apply(value));
45  }
46 };
