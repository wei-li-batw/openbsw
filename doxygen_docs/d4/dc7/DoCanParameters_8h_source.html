221 inline uint32_t DoCanParameters::decodeMinSeparationTime(uint8_t const encodedMinSeparationTime)
222 {
223  if (encodedMinSeparationTime <= 0x7FU)
224  {
225  return static_cast<uint32_t>(encodedMinSeparationTime) * 1000U;
226  }
227 
228  if ((encodedMinSeparationTime >= 0xF1U) && (encodedMinSeparationTime <= 0xF9U))
229  {
230  return (static_cast<uint32_t>(encodedMinSeparationTime) - 0xF0U) * 100U;
231  }
232 
233  return 0x7fU * 1000U;
234 }
235 
236 inline uint8_t DoCanParameters::encodeMinSeparationTime(uint32_t const minSeparationTimeUs)
237 {
238  // Encode 100-900us STmin values
239  if ((minSeparationTimeUs >= 100U) && (minSeparationTimeUs < 1000U))
240  {
241  return static_cast<uint8_t>(minSeparationTimeUs / 100U) + 0xF0U;
242  }
243 
244  // Encode millisecond STmin values
245  if (minSeparationTimeUs < (0x7FU * 1000U))
246  {
247  return static_cast<uint8_t>(minSeparationTimeUs / 1000U);
248  }
249 
250  // Default to max valid STmin for all other cases
251  return 0x7FU;
252 }
