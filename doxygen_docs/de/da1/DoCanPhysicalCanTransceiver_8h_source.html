132 template<class Addressing>
133 SendResult DoCanPhysicalCanTransceiver<Addressing>::startSendDataFrames(
134  FrameCodecType const& codec,
135  IDoCanDataFrameTransmitterCallback<DataLinkLayerType>& callback,
136  JobHandleType const jobHandle,
137  DataLinkAddressType const transmissionAddress,
138  FrameIndexType const firstFrameIndex,
139  FrameIndexType const lastFrameIndex,
140  FrameSizeType const consecutiveFrameDataSize,
141  ::etl::span<uint8_t const> const& data)
142 {
143  (void)lastFrameIndex;
144  if (!_sendPending)
145  {
146  ::etl::span<uint8_t> payload(
147  _frame.getPayload(), static_cast<size_t>(_frame.getMaxPayloadLength()));
148 
149  if (codec.encodeDataFrame(
150  payload, data, firstFrameIndex, consecutiveFrameDataSize, _sendDataSize)
151  != CodecResult::OK)
152  {
153  return SendResult::INVALID;
154  }
155 
156  uint32_t canId;
157  _addressing.encodeTransmissionAddress(transmissionAddress, canId, payload);
158  _frame.setId(canId);
159  _frame.setPayloadLength(static_cast<uint8_t>(payload.size()));
160  ::can::ICanTransceiver::ErrorCode const result = _transceiver.write(_frame, *this);
161  if (result == ::can::ICanTransceiver::ErrorCode::CAN_ERR_OK)
162  {
163  _sendPending = true;
164  _sendCallback = &callback;
165  _sendJobHandle = jobHandle;
166  return SendResult::QUEUED_FULL;
167  }
168  if (result != ::can::ICanTransceiver::ErrorCode::CAN_ERR_TX_HW_QUEUE_FULL)
169  {
170  return SendResult::FAILED;
171  }
172  // default return value: FULL
173  }
174  return SendResult::FULL;
175 }
